```python
import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc

ScalarType = PETSc.ScalarType


def solve(case_spec: dict) -> dict:
    comm = MPI.COMM_WORLD

    # Parameters
    epsilon = 0.05
    beta = [4.0, 0.0]
    
    # Mesh resolution - need fine enough for accuracy but fast enough
    nx, ny = 80, 80
    degree = 2

    domain = mesh.create_unit_square(comm, nx, ny, cell_type=mesh.CellType.triangle)
    V = fem.functionspace(domain, ("Lagrange", degree))

    # Spatial coordinate
    x = ufl.SpatialCoordinate(domain)

    # Exact solution
    u_exact_ufl = ufl.exp(2 * x[0]) * ufl.sin(ufl.pi * x[1])

    # Compute source term from manufactured solution
    # -eps * laplacian(u) + beta . grad(u) = f
    # u = exp(2x)*sin(pi*y)
    # grad(u) = [2*exp(2x)*sin(pi*y), exp(2x)*pi*cos(pi*y)]
    # laplacian(u) = 4*exp(2x)*sin(pi*y) - pi^2*exp(2x)*sin(pi*y) = (4 - pi^2)*exp(2x)*sin(pi*y)
    # f = -eps*(4-pi^2)*exp(2x)*sin(pi*y) + 4*2*exp(2x)*sin(pi*y)
    # f = exp(2x)*sin(pi*y) * [-eps*(4-pi^2) + 8]
    
    # Use UFL to compute f symbolically
    grad_u_exact = ufl.grad(u_exact_ufl)
    laplacian_u_exact = ufl.div(ufl.grad(u_exact_ufl))
    beta_ufl = ufl.as_vector([fem.Constant(domain, ScalarType(beta[0])),
                               fem.Constant(domain, ScalarType(beta[1]))])
    eps_const = fem.Constant(domain, ScalarType(epsilon))
    
    f_ufl = -eps_const * laplacian_u_exact + ufl.dot(beta_ufl, grad_u_exact)

    # Trial and test functions
    u = ufl.TrialFunction(V)
    v = ufl.TestFunction(V)

    # Standard Galerkin
    a = eps_const * ufl.inner(ufl.grad(u), ufl.grad(v)) * ufl.dx + ufl.dot(beta_ufl, ufl.grad(u)) * v * ufl.dx
    L = f_ufl * v * ufl.dx

    # SUPG stabilization
    h = ufl.CellDiameter(domain)
    beta_norm = ufl.sqrt(ufl.dot(beta_ufl, beta_ufl))
    Pe_cell = beta_norm * h / (2.0 * eps_const)
    # tau = h / (2 * |beta|) * (coth(Pe) - 1/Pe) ≈ h/(2|beta|) for large Pe
    # Simplified tau for high Peclet
    tau = h / (2.0 * beta_norm) * (ufl.conditional(ufl.gt(Pe_cell, 1.0), 1.0 - 1.0/Pe_cell, Pe_cell/3.0))
    
    # SUPG residual: R(u) = -eps*laplacian(u) + beta.grad(u) - f
    # For linear elements, laplacian vanishes; for higher order it doesn't but we approximate
    # The strong residual applied to trial function:
    # For the bilinear form, the SUPG test function modification is: v_supg = tau * beta . grad(v)
    r_supg = ufl.dot(beta_ufl, ufl.grad(v))
    
    a_supg = tau * ufl.dot(beta_ufl, ufl.grad(u)) * r_supg * ufl.dx
    # For degree >= 2, include diffusion part of strong residual (but often small contribution)
    # We skip -eps*laplacian(u) in SUPG for simplicity (it's second order and small for high Pe)
    L_supg = tau * f_ufl * r_supg * ufl.dx

    a_total = a + a_supg
    L_total = L + L_supg

    # Boundary conditions
    tdim = domain.topology.dim
    fdim = tdim - 1

    # All boundary
    def all_boundary(x):
        return (np.isclose(x[0], 0.0) | np.isclose(x[0], 1.0) |
                np.isclose(x[1], 0.0) | np.isclose(x[1], 1.0))

    boundary_facets = mesh.locate_entities_boundary(domain, fdim, all_boundary)
    dofs = fem.locate_dofs_topological(V, fdim, boundary_facets)

    u_bc = fem.Function(V)
    u_bc.interpolate(lambda x: np.exp(2 * x[0]) * np.sin(np.pi * x[1]))
    bc = fem.dirichletbc(u_bc, dofs)

    # Solve
    problem = petsc.LinearProblem(
        a_total, L_total, bcs=[bc],
        petsc_options={
            "ksp_type": "gmres",
            "pc_type": "ilu",
            "ksp_rtol": "1e-10",
            "ksp_atol": "1e-12",
            "ksp_max_it": "2000",
        },
        petsc_options_prefix="convdiff_"
    )
    u_sol = problem.solve()

    # Get iteration count
    ksp = problem.solver
    iterations = ksp.getIterationNumber()

    # Evaluate on 50x50 grid
    nx_eval, ny_eval = 50, 50
    xs = np.linspace(0, 1, nx_eval)
    ys = np.linspace(0, 1, ny_eval)
    X, Y = np.meshgrid(xs, ys, indexing='ij')
    points = np.zeros((3, nx_eval * ny_eval))
    points[0] = X.ravel()
    points[1] = Y.ravel()

    bb_tree = geometry.bb_tree(domain, domain.topology.dim)
    cell_candidates = geometry.compute_collisions_points(bb_tree, points.T)
    colliding_cells = geometry.compute_colliding_cells(domain, cell_candidates, points.T)

    points_on_proc = []
    cells_on_proc = []
    eval_map = []
    for i in range(points.shape[1]):
        links = colliding_cells.links(i)
        if len(links) > 0:
            points_on_proc.append(points[:, i])
            cells_on_proc.append(links[0])
            eval_map.append(i)

    u_values = np.full(nx_eval * ny_eval, np.nan)
    if len(points_on_proc) > 0:
        pts_arr = np.array(points_on_proc)
        cells_arr = np.array(cells_on_proc, dtype=np.int32)
        vals = u_sol.eval(pts_arr, cells_arr)
        u_values[eval_map] = vals.flatten()

    u_grid = u_values.reshape((nx_eval, ny_eval))

    solver_info = {
        "mesh_resolution": nx,
        "element_degree": degree,
        "ksp_type": "gmres",
        "pc_type": "ilu",
        "rtol": 1e-10,
        "iterations": iterations,
    }

    return {"u": u_grid, "solver_info": solver_info}
```
