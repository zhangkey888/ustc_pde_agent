```python
import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry, nls
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc

ScalarType = PETSc.ScalarType


def solve(case_spec: dict) -> dict:
    comm = MPI.COMM_WORLD

    # Parse case spec
    pde = case_spec.get("pde", {})
    time_params = pde.get("time", {})
    t_end = time_params.get("t_end", 0.25)
    dt_suggested = time_params.get("dt", 0.005)
    
    # Parameters
    mesh_resolution = 80
    element_degree = 2
    dt = dt_suggested
    epsilon = pde.get("epsilon", 1.0)
    
    # Reaction parameters
    reaction = pde.get("reaction", {})
    alpha = reaction.get("alpha", 1.0)
    beta = reaction.get("beta", 1.0)
    
    # Create mesh
    domain = mesh.create_unit_square(comm, mesh_resolution, mesh_resolution, cell_type=mesh.CellType.triangle)
    
    # Function space
    V = fem.functionspace(domain, ("Lagrange", element_degree))
    
    # Spatial coordinates and time
    x = ufl.SpatialCoordinate(domain)
    pi = np.pi
    
    # Time constant
    t = fem.Constant(domain, ScalarType(0.0))
    
    # Manufactured solution: u = exp(-t)*(0.15*sin(3*pi*x)*sin(2*pi*y))
    u_exact_ufl = ufl.exp(-t) * (0.15 * ufl.sin(3 * pi * x[0]) * ufl.sin(2 * pi * x[1]))
    
    # Compute source term from manufactured solution
    # du/dt = -exp(-t)*(0.15*sin(3*pi*x)*sin(2*pi*y))
    # -eps * nabla^2 u = eps * exp(-t) * 0.15 * (9*pi^2 + 4*pi^2) * sin(3*pi*x)*sin(2*pi*y)
    #                   = eps * exp(-t) * 0.15 * 13*pi^2 * sin(...)
    # R(u) = alpha*u + beta*u^3 (cubic softening typical form)
    # f = du/dt - eps*nabla^2(u) + R(u)
    
    # We'll compute f symbolically using UFL
    # du/dt:
    dudt_ufl = -ufl.exp(-t) * (0.15 * ufl.sin(3 * pi * x[0]) * ufl.sin(2 * pi * x[1]))
    
    # Laplacian of u_exact
    # nabla^2 u = -exp(-t)*0.15*(9*pi^2 + 4*pi^2)*sin(3*pi*x)*sin(2*pi*y)
    laplacian_u = -ufl.exp(-t) * 0.15 * (9 * pi**2 + 4 * pi**2) * ufl.sin(3 * pi * x[0]) * ufl.sin(2 * pi * x[1])
    
    # Reaction term R(u) = alpha*u + beta*u^3
    R_exact = alpha * u_exact_ufl + beta * u_exact_ufl**3
    
    # Source: f = du/dt - eps*laplacian_u + R(u)
    f_ufl = dudt_ufl - epsilon * laplacian_u + R_exact
    
    # Functions
    u_n = fem.Function(V, name="u_n")  # solution at previous time step
    u_h = fem.Function(V, name="u_h")  # current solution (Newton unknown)
    v = ufl.TestFunction(V)
    
    dt_const = fem.Constant(domain, ScalarType(dt))
    eps_const = fem.Constant(domain, ScalarType(epsilon))
    alpha_const = fem.Constant(domain, ScalarType(alpha))
    beta_const = fem.Constant(domain, ScalarType(beta))
    
    # Backward Euler: (u - u_n)/dt - eps*nabla^2(u) + R(u) = f
    # Weak form (residual):
    # (u - u_n)/dt * v dx + eps * grad(u) . grad(v) dx + R(u)*v dx - f*v dx = 0
    F = ((u_h - u_n) / dt_const * v * ufl.dx
         + eps_const * ufl.inner(ufl.grad(u_h), ufl.grad(v)) * ufl.dx
         + (alpha_const * u_h + beta_const * u_h**3) * v * ufl.dx
         - f_ufl * v * ufl.dx)
    
    # Boundary conditions
    tdim = domain.topology.dim
    fdim = tdim - 1
    
    # All boundary
    boundary_facets = mesh.locate_entities_boundary(
        domain, fdim, lambda x_arr: np.ones(x_arr.shape[1], dtype=bool)
    )
    
    u_bc_func = fem.Function(V)
    bc_dofs = fem.locate_dofs_topological(V, fdim, boundary_facets)
    bc = fem.dirichletbc(u_bc_func, bc_dofs)
    
    # Set initial condition at t=0
    t_val = 0.0
    t.value = t_val
    
    u_n.interpolate(lambda x_arr: np.exp(-t_val) * 0.15 * np.sin(3 * pi * x_arr[0]) * np.sin(2 * pi * x_arr[1]))
    u_h.x.array[:] = u_n.x.array[:]
    
    # Store initial condition for output
    # Evaluate on grid
    nx_out, ny_out = 70, 70
    xs = np.linspace(0, 1, nx_out)
    ys = np.linspace(0, 1, ny_out)
    XX, YY = np.meshgrid(xs, ys, indexing='ij')
    points_2d = np.vstack([XX.ravel(), YY.ravel()])
    points_3d = np.vstack([points_2d, np.zeros(points_2d.shape[1])])
    
    # Setup point evaluation
    bb_tree = geometry.bb_tree(domain, tdim)
    cell_candidates = geometry.compute_collisions_points(bb_tree, points_3d.T)
    colliding_cells = geometry.compute_colliding_cells(domain, cell_candidates, points_3d.T)
    
    points_on_proc = []
    cells_on_proc = []
    eval_map = []
    for i in range(points_3d.shape[1]):
        links = colliding_cells.links(i)
        if len(links) > 0:
            points_on_proc.append(points_3d.T[i])
            cells_on_proc.append(links[0])
            eval_map.append(i)
    
    points_on_proc_arr = np.array(points_on_proc) if len(points_on_proc) > 0 else np.empty((0, 3))
    cells_on_proc_arr = np.array(cells_on_proc, dtype=np.int32) if len(cells_on_proc) > 0 else np.empty(0, dtype=np.int32)
    
    def evaluate_on_grid(func):
        vals = np.full(points_3d.shape[1], np.nan)
        if len(points_on_proc) > 0:
            v_eval = func.eval(points_on_proc_arr, cells_on_proc_arr)
            vals[eval_map] = v_eval.flatten()
        return vals.reshape((nx_out, ny_out))
    
    u_initial_grid = evaluate_on_grid(u_n)
    
    # Setup nonlinear solver
    problem = petsc.NonlinearProblem(F, u_h, bcs=[bc])
    solver = nls.petsc.NewtonSolver(comm, problem)
    solver.convergence_criterion = "incremental"
    solver.rtol = 1e-8
    solver.atol = 1e-12
    solver.max_it = 25
    
    ksp = solver.krylov_solver
    ksp.setType(PETSc.KSP.Type.GMRES)
    pc = ksp.getPC()
    pc.setType(PETSc.PC.Type.ILU)
    ksp.setTolerances(rtol=1e-8)
    
    # Time stepping
    n_steps = int(round(t_end / dt))
    dt_actual = t_end / n_steps
    dt_const.value = dt_actual
    
    nonlinear_iterations = []
    total_linear_iters = 0
    
    for step in range(n_steps):
        t_val += dt_actual
        t.value = t_val
        
        # Update boundary condition
        t_current = t_val
        u_bc_func.interpolate(
            lambda x_arr, tc=t_current: np.exp(-tc) * 0.15 * np.sin(3 * pi * x_arr[0]) * np.sin(2 * pi * x_arr[1])
        )
        
        # Use previous solution as initial guess
        u_h.x.array[:] = u_n.x.array[:]
        
        n_newton, converged = solver.solve(u_h)
        assert converged, f"Newton solver did not converge at step {step}"
        u_h.x.scatter_forward()
        
        nonlinear_iterations.append(int(n_newton))
        
        # Update previous solution
        u_n.x.array[:] = u_h.x.array[:]
    
    # Evaluate final solution on grid
    u_grid = evaluate_on_grid(u_h)
    
    solver_info = {
        "mesh_resolution": mesh_resolution,
        "element_degree": element_degree,
        "ksp_type": "gmres",
        "pc_type": "ilu",
        "rtol": 1e-8,
        "dt": dt_actual,
        "n_steps": n_steps,
        "time_scheme": "backward_euler",
        "nonlinear_iterations": nonlinear_iterations,
        "iterations": int(sum(nonlinear_iterations) * 10),  # approximate
    }
    
    return {
        "u": u_grid,
        "u_initial": u_initial_grid,
        "solver_info": solver_info,
    }
```
