```python
import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc

ScalarType = PETSc.ScalarType


def solve(case_spec: dict) -> dict:
    comm = MPI.COMM_WORLD
    
    # Parameters
    nu_val = 0.8
    nx = ny = 64
    degree_u = 2
    degree_p = 1
    
    # Create mesh
    domain = mesh.create_unit_square(comm, nx, ny, cell_type=mesh.CellType.triangle)
    tdim = domain.topology.dim
    fdim = tdim - 1
    
    # Function spaces (Taylor-Hood P2/P1)
    V = fem.functionspace(domain, ("Lagrange", degree_u, (domain.geometry.dim,)))
    Q = fem.functionspace(domain, ("Lagrange", degree_p))
    
    # Mixed function space
    mel = ufl.MixedElement([V.ufl_element(), Q.ufl_element()])
    W = fem.functionspace(domain, mel)
    
    # Trial and test functions
    (u, p) = ufl.TrialFunctions(W)
    (v, q) = ufl.TestFunctions(W)
    
    # Source term
    f = fem.Constant(domain, np.array([0.0, 0.0], dtype=ScalarType))
    nu = fem.Constant(domain, ScalarType(nu_val))
    
    # Bilinear form for Stokes
    a = (nu * ufl.inner(ufl.grad(u), ufl.grad(v)) * ufl.dx
         - p * ufl.div(v) * ufl.dx
         + q * ufl.div(u) * ufl.dx)
    
    L = ufl.inner(f, v) * ufl.dx
    
    # Boundary conditions
    # Parse boundary conditions from case_spec
    bc_specs = case_spec.get("bcs", [])
    
    bcs = []
    
    for bc_spec in bc_specs:
        bc_type = bc_spec.get("type", "")
        location = bc_spec.get("location", "")
        value = bc_spec.get("value", None)
        
        if bc_type == "dirichlet":
            # Determine boundary marker
            if location == "left":
                def marker(x, loc=location):
                    return np.isclose(x[0], 0.0)
            elif location == "right":
                def marker(x, loc=location):
                    return np.isclose(x[0], 1.0)
            elif location == "bottom":
                def marker(x, loc=location):
                    return np.isclose(x[1], 0.0)
            elif location == "top":
                def marker(x, loc=location):
                    return np.isclose(x[1], 1.0)
            elif location == "all":
                def marker(x, loc=location):
                    return (np.isclose(x[0], 0.0) | np.isclose(x[0], 1.0) |
                            np.isclose(x[1], 0.0) | np.isclose(x[1], 1.0))
            else:
                continue
            
            facets = mesh.locate_entities_boundary(domain, fdim, marker)
            
            # Parse value
            if isinstance(value, (list, tuple)):
                val_strs = value
            elif isinstance(value, str):
                val_strs = [value, "0.0"]
            else:
                val_strs = ["0.0", "0.0"]
            
            # Create interpolation function
            u_bc_func = fem.Function(V)
            
            # Build lambda for interpolation
            def make_interp(val_strs):
                def interp_func(x):
                    result = np.zeros((domain.geometry.dim, x.shape[1]))
                    for i, vs in enumerate(val_strs):
                        if i < domain.geometry.dim:
                            try:
                                # Try to evaluate as expression
                                result[i] = eval(vs, {"np": np, "x": x, "pi": np.pi,
                                                       "__builtins__": {}})
                            except:
                                try:
                                    result[i] = float(vs)
                                except:
                                    result[i] = 0.0
                    return result
                return interp_func
            
            u_bc_func.interpolate(make_interp(val_strs))
            
            dofs = fem.locate_dofs_topological((W.sub(0), V), fdim, facets)
            bc = fem.dirichletbc(u_bc_func, dofs, W.sub(0))
            bcs.append(bc)
    
    # If no BCs parsed, set up default no-slip on all boundaries
    if len(bcs) == 0:
        # Default: no-slip everywhere
        all_facets = mesh.locate_entities_boundary(
            domain, fdim, lambda x: np.ones(x.shape[1], dtype=bool))
        u_zero = fem.Function(V)
        u_zero.interpolate(lambda x: np.zeros((domain.geometry.dim, x.shape[1])))
        dofs = fem.locate_dofs_topological((W.sub(0), V), fdim, all_facets)
        bc = fem.dirichletbc(u_zero, dofs, W.sub(0))
        bcs.append(bc)
    
    # Pin pressure at one point to remove nullspace
    # Find a point (e.g., bottom-left corner) and pin pressure there
    def bottom_left(x):
        return np.isclose(x[0], 0.0) & np.isclose(x[1], 0.0)
    
    # Try to pin pressure
    try:
        p_dofs = fem.locate_dofs_geometrical((W.sub(1), Q), bottom_left)
        p_zero = fem.Function(Q)
        p_zero.x.array[:] = 0.0
        bc_p = fem.dirichletbc(p_zero, p_dofs, W.sub(1))
        bcs.append(bc_p)
    except:
        pass
    
    # Solve
    ksp_type = "gmres"  # changed from minres for better robustness with saddle-point
    pc_type = "ilu"
    rtol = 1e-10
    
    problem = petsc.LinearProblem(
        a, L, bcs=bcs,
        petsc_options={
            "ksp_type": ksp_type,
            "pc_type": pc_type,
            "ksp_rtol": str(rtol),
            "ksp_max_it": "5000",
            "ksp_monitor": None,
        },
        petsc_options_prefix="stokes_"
    )
    
    wh = problem.solve()
    wh.x.scatter_forward()
    
    # Get iteration count
    try:
        iterations = problem.solver.getIterationNumber()
    except:
        iterations = -1
    
    # Extract velocity sub-function
    u_sol = wh.sub(0).collapse()
    
    # Sample on 100x100 grid
    nx_out, ny_out = 100, 100
    xs = np.linspace(0, 1, nx_out)
    ys = np.linspace(0, 1, ny_out)
    XX, YY = np.meshgrid(xs, ys, indexing='ij')
    points = np.zeros((3, nx_out * ny_out))
    points[0] = XX.ravel()
    points[1] = YY.ravel()
    
    # Point evaluation
    bb_tree = geometry.bb_tree(domain, tdim)
    cell_candidates = geometry.compute_collisions_points(bb_tree, points.T)
    colliding_cells = geometry.compute_colliding_cells(domain, cell_candidates, points.T)
    
    vel_mag = np.full(nx_out * ny_out, np.nan)
    
    points_on_proc = []
    cells_on_proc = []
    eval_map = []
    for i in range(nx_out * ny_out):
        links = colliding_cells.links(i)
        if len(links) > 0:
            points_on_proc.append(points.T[i])
            cells_on_proc.append(links[0])
            eval_map.append(i)
    
    if len(points_on_proc) > 0:
        pts_arr = np.array(points_on_proc)
        cells_arr = np.array(cells_on_proc, dtype=np.int32)
        vals = u_sol.eval(pts_arr, cells_arr)
        # vals shape: (n_points, gdim)
        for idx, global_idx in enumerate(eval_map):
            ux = vals[idx, 0]
            uy = vals[idx, 1]
            vel_mag[global_idx] = np.sqrt(ux**2 + uy**2)
    
    u_grid = vel_mag.reshape((nx_out, ny_out))
    
    # Fill any NaN with nearest neighbor or 0
    if np.any(np.isnan(u_grid)):
        u_grid = np.nan_to_num(u_grid, nan=0.0)
    
    solver_info = {
        "mesh_resolution": nx,
        "element_degree": degree_u,
        "ksp_type": ksp_type,
        "pc_type": pc_type,
        "rtol": rtol,
        "iterations": int(iterations),
    }
    
    return {
        "u": u_grid,
        "solver_info": solver_info,
    }
```
