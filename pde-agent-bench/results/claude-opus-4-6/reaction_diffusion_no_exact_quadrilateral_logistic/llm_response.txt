import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry, nls
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc


def solve(case_spec: dict) -> dict:
    comm = MPI.COMM_WORLD
    
    # Parse case spec
    pde = case_spec.get("pde", {})
    domain_spec = case_spec.get("domain", {})
    time_spec = pde.get("time", {})
    
    # Parameters
    epsilon = pde.get("epsilon", 1.0)
    rho = pde.get("reaction_rho", 1.0)  # logistic growth rate
    f_val = pde.get("source_term", 1.0)
    
    # Time parameters
    t_end = time_spec.get("t_end", 0.3)
    dt_val = time_spec.get("dt", 0.01)
    scheme = time_spec.get("scheme", "backward_euler")
    
    # Mesh
    nx = 65
    ny = 65
    
    domain_obj = mesh.create_unit_square(
        comm, nx, ny, cell_type=mesh.CellType.quadrilateral
    )
    
    # Function space - degree 2 for better accuracy
    degree = 2
    V = fem.functionspace(domain_obj, ("Lagrange", degree))
    
    # Functions
    u = fem.Function(V)
    u.name = "u"
    u_n = fem.Function(V)  # previous time step
    v = ufl.TestFunction(V)
    
    # Spatial coordinate
    x = ufl.SpatialCoordinate(domain_obj)
    
    # Source term
    f = fem.Constant(domain_obj, PETSc.ScalarType(f_val))
    
    # Diffusion coefficient
    eps_c = fem.Constant(domain_obj, PETSc.ScalarType(epsilon))
    
    # Logistic reaction rate
    rho_c = fem.Constant(domain_obj, PETSc.ScalarType(rho))
    
    # Time step
    dt_c = fem.Constant(domain_obj, PETSc.ScalarType(dt_val))
    
    # Initial condition: u0 = 0.25 + 0.15*sin(pi*x)*sin(pi*y)
    u_n.interpolate(lambda X: 0.25 + 0.15 * np.sin(np.pi * X[0]) * np.sin(np.pi * X[1]))
    u.interpolate(lambda X: 0.25 + 0.15 * np.sin(np.pi * X[0]) * np.sin(np.pi * X[1]))
    
    # Store initial condition for output
    u_init = fem.Function(V)
    u_init.interpolate(lambda X: 0.25 + 0.15 * np.sin(np.pi * X[0]) * np.sin(np.pi * X[1]))
    
    # Boundary conditions - use initial condition values on boundary (or zero flux)
    # For reaction-diffusion with no exact solution specified, use Dirichlet from initial condition
    # Actually, let's check if there's a BC specification
    bc_spec = pde.get("boundary_conditions", None)
    
    tdim = domain_obj.topology.dim
    fdim = tdim - 1
    
    # Dirichlet BC: g on boundary
    # Default: use the initial condition on the boundary (which is what makes sense for this problem)
    def boundary(x):
        return (np.isclose(x[0], 0.0) | np.isclose(x[0], 1.0) |
                np.isclose(x[1], 0.0) | np.isclose(x[1], 1.0))
    
    boundary_facets = mesh.locate_entities_boundary(domain_obj, fdim, boundary)
    boundary_dofs = fem.locate_dofs_topological(V, fdim, boundary_facets)
    
    # BC function - we'll update this each time step if needed
    # For this problem, the BC is typically the initial condition on the boundary
    # or zero Dirichlet. Let's use the IC values on boundary.
    u_bc = fem.Function(V)
    u_bc.interpolate(lambda X: 0.25 + 0.15 * np.sin(np.pi * X[0]) * np.sin(np.pi * X[1]))
    bc = fem.dirichletbc(u_bc, boundary_dofs)
    bcs = [bc]
    
    # Nonlinear residual: backward Euler
    # du/dt - eps*laplacian(u) + R(u) = f
    # R(u) = rho * u * (1 - u)  (logistic reaction)
    # Backward Euler: (u - u_n)/dt - eps*laplacian(u) + rho*u*(1-u) = f
    
    F = (
        ufl.inner((u - u_n) / dt_c, v) * ufl.dx
        + eps_c * ufl.inner(ufl.grad(u), ufl.grad(v)) * ufl.dx
        + rho_c * u * (1.0 - u) * v * ufl.dx
        - f * v * ufl.dx
    )
    
    # Setup nonlinear solver
    problem = petsc.NonlinearProblem(F, u, bcs=bcs)
    solver = nls.petsc.NewtonSolver(comm, problem)
    solver.convergence_criterion = "incremental"
    solver.rtol = 1e-8
    solver.atol = 1e-10
    solver.max_it = 25
    solver.report = False
    
    ksp = solver.krylov_solver
    ksp.setType(PETSc.KSP.Type.GMRES)
    pc = ksp.getPC()
    pc.setType(PETSc.PC.Type.ILU)
    ksp.setTolerances(rtol=1e-8)
    
    # Time stepping
    t = 0.0
    n_steps = int(np.round(t_end / dt_val))
    nonlinear_iterations = []
    total_linear_iterations = 0
    
    for step in range(n_steps):
        t += dt_val
        
        n_iter, converged = solver.solve(u)
        assert converged, f"Newton solver did not converge at step {step}, t={t}"
        u.x.scatter_forward()
        
        nonlinear_iterations.append(int(n_iter))
        # Approximate linear iterations (each Newton step ~ 1 linear solve)
        total_linear_iterations += int(n_iter)
        
        # Update previous solution
        u_n.x.array[:] = u.x.array[:]
    
    # Evaluate on 65x65 grid
    nx_eval = 65
    ny_eval = 65
    xs = np.linspace(0.0, 1.0, nx_eval)
    ys = np.linspace(0.0, 1.0, ny_eval)
    XX, YY = np.meshgrid(xs, ys, indexing='ij')
    
    points = np.zeros((3, nx_eval * ny_eval))
    points[0, :] = XX.ravel()
    points[1, :] = YY.ravel()
    
    # Point evaluation
    bb_tree = geometry.bb_tree(domain_obj, domain_obj.topology.dim)
    cell_candidates = geometry.compute_collisions_points(bb_tree, points.T)
    colliding_cells = geometry.compute_colliding_cells(domain_obj, cell_candidates, points.T)
    
    points_on_proc = []
    cells_on_proc = []
    eval_map = []
    for i in range(points.shape[1]):
        links = colliding_cells.links(i)
        if len(links) > 0:
            points_on_proc.append(points[:, i])
            cells_on_proc.append(links[0])
            eval_map.append(i)
    
    u_values = np.full(nx_eval * ny_eval, np.nan)
    if len(points_on_proc) > 0:
        pts_arr = np.array(points_on_proc)
        cells_arr = np.array(cells_on_proc, dtype=np.int32)
        vals = u.eval(pts_arr, cells_arr)
        u_values[eval_map] = vals.flatten()
    
    u_grid = u_values.reshape((nx_eval, ny_eval))
    
    # Also evaluate initial condition on same grid
    u_init_values = np.full(nx_eval * ny_eval, np.nan)
    if len(points_on_proc) > 0:
        vals_init = u_init.eval(pts_arr, cells_arr)
        u_init_values[eval_map] = vals_init.flatten()
    u_init_grid = u_init_values.reshape((nx_eval, ny_eval))
    
    return {
        "u": u_grid,
        "u_initial": u_init_grid,
        "solver_info": {
            "mesh_resolution": nx,
            "element_degree": degree,
            "ksp_type": "gmres",
            "pc_type": "ilu",
            "rtol": 1e-8,
            "iterations": total_linear_iterations,
            "dt": dt_val,
            "n_steps": n_steps,
            "time_scheme": "backward_euler",
            "nonlinear_iterations": nonlinear_iterations,
        }
    }
