import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc

ScalarType = PETSc.ScalarType


def solve(case_spec: dict) -> dict:
    comm = MPI.COMM_WORLD
    
    # Parameters
    N = 80
    degree = 2
    ksp_type = "cg"
    pc_type = "hypre"
    rtol = 1e-10
    
    # Create mesh
    domain = mesh.create_unit_square(comm, N, N, cell_type=mesh.CellType.triangle)
    
    # Function spaces
    V = fem.functionspace(domain, ("Lagrange", degree))
    
    # Manufactured solution: u = tanh(6*(y-0.5))*sin(pi*x)
    x = ufl.SpatialCoordinate(domain)
    pi = ufl.pi
    
    u_exact_ufl = ufl.tanh(6.0 * (x[1] - 0.5)) * ufl.sin(pi * x[0])
    
    # Compute the source term f = Δ²u
    # We need the biharmonic of u_exact
    grad_u = ufl.grad(u_exact_ufl)
    lap_u = ufl.div(grad_u)
    grad_lap_u = ufl.grad(lap_u)
    bilap_u = ufl.div(grad_lap_u)
    
    f_ufl = bilap_u
    
    # Mixed formulation: introduce w = -Δu
    # System:
    #   -Δu = w  in Ω,  u = g on ∂Ω
    #   -Δw = f  in Ω,  w = -Δu_exact on ∂Ω
    
    # Compute w_exact = -Δu_exact for boundary condition on w
    w_exact_ufl = -lap_u
    
    # Boundary conditions
    tdim = domain.topology.dim
    fdim = tdim - 1
    
    # All boundary facets
    domain.topology.create_connectivity(fdim, tdim)
    boundary_facets = mesh.exterior_facet_indices(domain.topology)
    
    # BC for u
    u_bc_func = fem.Function(V)
    u_exact_expr = fem.Expression(u_exact_ufl, V.element.interpolation_points)
    u_bc_func.interpolate(u_exact_expr)
    
    dofs_u = fem.locate_dofs_topological(V, fdim, boundary_facets)
    bc_u = fem.dirichletbc(u_bc_func, dofs_u)
    
    # BC for w
    w_bc_func = fem.Function(V)
    w_exact_expr = fem.Expression(w_exact_ufl, V.element.interpolation_points)
    w_bc_func.interpolate(w_exact_expr)
    
    dofs_w = fem.locate_dofs_topological(V, fdim, boundary_facets)
    bc_w = fem.dirichletbc(w_bc_func, dofs_w)
    
    # Source term as a Function
    f_func = fem.Function(V)
    f_expr = fem.Expression(f_ufl, V.element.interpolation_points)
    f_func.interpolate(f_expr)
    
    total_iterations = 0
    
    # Step 1: Solve -Δw = f with w = w_exact on ∂Ω
    w_sol = fem.Function(V)
    trial_w = ufl.TrialFunction(V)
    test_v = ufl.TestFunction(V)
    
    a_w = ufl.inner(ufl.grad(trial_w), ufl.grad(test_v)) * ufl.dx
    L_w = ufl.inner(f_func, test_v) * ufl.dx
    
    problem_w = petsc.LinearProblem(
        a_w, L_w, bcs=[bc_w],
        petsc_options={
            "ksp_type": ksp_type,
            "pc_type": pc_type,
            "ksp_rtol": str(rtol),
            "ksp_max_it": "2000",
        },
        petsc_options_prefix="step1_"
    )
    w_sol = problem_w.solve()
    
    # Get iteration count
    ksp1 = problem_w.solver
    total_iterations += ksp1.getIterationNumber()
    
    # Step 2: Solve -Δu = w with u = g on ∂Ω
    u_sol = fem.Function(V)
    trial_u = ufl.TrialFunction(V)
    
    a_u = ufl.inner(ufl.grad(trial_u), ufl.grad(test_v)) * ufl.dx
    L_u = ufl.inner(w_sol, test_v) * ufl.dx
    
    problem_u = petsc.LinearProblem(
        a_u, L_u, bcs=[bc_u],
        petsc_options={
            "ksp_type": ksp_type,
            "pc_type": pc_type,
            "ksp_rtol": str(rtol),
            "ksp_max_it": "2000",
        },
        petsc_options_prefix="step2_"
    )
    u_sol = problem_u.solve()
    
    ksp2 = problem_u.solver
    total_iterations += ksp2.getIterationNumber()
    
    # Evaluate on 50x50 grid
    nx_out, ny_out = 50, 50
    xs = np.linspace(0, 1, nx_out)
    ys = np.linspace(0, 1, ny_out)
    XX, YY = np.meshgrid(xs, ys, indexing='ij')
    
    points = np.zeros((3, nx_out * ny_out))
    points[0, :] = XX.ravel()
    points[1, :] = YY.ravel()
    
    bb_tree = geometry.bb_tree(domain, domain.topology.dim)
    cell_candidates = geometry.compute_collisions_points(bb_tree, points.T)
    colliding_cells = geometry.compute_colliding_cells(domain, cell_candidates, points.T)
    
    points_on_proc = []
    cells_on_proc = []
    eval_map = []
    for i in range(points.shape[1]):
        links = colliding_cells.links(i)
        if len(links) > 0:
            points_on_proc.append(points.T[i])
            cells_on_proc.append(links[0])
            eval_map.append(i)
    
    u_values = np.full(nx_out * ny_out, np.nan)
    if len(points_on_proc) > 0:
        vals = u_sol.eval(np.array(points_on_proc), np.array(cells_on_proc, dtype=np.int32))
        u_values[eval_map] = vals.flatten()
    
    u_grid = u_values.reshape((nx_out, ny_out))
    
    return {
        "u": u_grid,
        "solver_info": {
            "mesh_resolution": N,
            "element_degree": degree,
            "ksp_type": ksp_type,
            "pc_type": pc_type,
            "rtol": rtol,
            "iterations": total_iterations,
        }
    }
