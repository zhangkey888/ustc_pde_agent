```python
import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc

ScalarType = PETSc.ScalarType


def solve(case_spec: dict) -> dict:
    comm = MPI.COMM_WORLD

    # Parameters
    epsilon = 0.01
    beta = np.array([12.0, 4.0])
    
    # Mesh resolution - P3 elements need fewer cells but we need accuracy <= 1e-6
    N = 64
    element_degree = 3

    domain = mesh.create_unit_square(comm, N, N, cell_type=mesh.CellType.triangle)
    
    V = fem.functionspace(domain, ("Lagrange", element_degree))
    
    # Manufactured solution: u = sin(pi*x)*sin(2*pi*y)
    x = ufl.SpatialCoordinate(domain)
    u_exact_ufl = ufl.sin(ufl.pi * x[0]) * ufl.sin(2 * ufl.pi * x[1])
    
    # Source term: f = -eps * laplacian(u) + beta . grad(u)
    # laplacian(u) = -pi^2 * sin(pi*x)*sin(2*pi*y) - 4*pi^2 * sin(pi*x)*sin(2*pi*y)
    #              = -5*pi^2 * sin(pi*x)*sin(2*pi*y)
    # So -eps * laplacian(u) = 5*eps*pi^2 * sin(pi*x)*sin(2*pi*y)
    # grad(u) = (pi*cos(pi*x)*sin(2*pi*y), 2*pi*sin(pi*x)*cos(2*pi*y))
    # beta . grad(u) = 12*pi*cos(pi*x)*sin(2*pi*y) + 4*2*pi*sin(pi*x)*cos(2*pi*y)
    
    grad_u_exact = ufl.grad(u_exact_ufl)
    laplacian_u_exact = ufl.div(ufl.grad(u_exact_ufl))
    
    beta_ufl = ufl.as_vector([fem.Constant(domain, ScalarType(beta[0])),
                               fem.Constant(domain, ScalarType(beta[1]))])
    eps_const = fem.Constant(domain, ScalarType(epsilon))
    
    f_expr = -eps_const * laplacian_u_exact + ufl.dot(beta_ufl, grad_u_exact)
    
    # Boundary conditions
    tdim = domain.topology.dim
    fdim = tdim - 1
    
    domain.topology.create_connectivity(fdim, tdim)
    boundary_facets = mesh.exterior_facets(domain.topology)
    
    u_bc = fem.Function(V)
    u_bc.interpolate(lambda x: np.sin(np.pi * x[0]) * np.sin(2 * np.pi * x[1]))
    
    dofs = fem.locate_dofs_topological(V, fdim, boundary_facets)
    bc = fem.dirichletbc(u_bc, dofs)
    
    # SUPG stabilization
    u = ufl.TrialFunction(V)
    v = ufl.TestFunction(V)
    
    # Element size
    h = ufl.CellDiameter(domain)
    
    # Peclet number based on element
    beta_norm = ufl.sqrt(ufl.dot(beta_ufl, beta_ufl))
    Pe_h = beta_norm * h / (2.0 * eps_const)
    
    # SUPG stabilization parameter
    # tau = h / (2 * |beta|) * (coth(Pe_h) - 1/Pe_h)
    # For high Pe, coth(Pe) - 1/Pe ≈ 1, so tau ≈ h / (2*|beta|)
    # Use a simpler formula that works well:
    tau = h / (2.0 * beta_norm) * (1.0 / ufl.tanh(Pe_h) - 1.0 / Pe_h)
    
    # Standard Galerkin
    a_gal = eps_const * ufl.inner(ufl.grad(u), ufl.grad(v)) * ufl.dx + \
            ufl.dot(beta_ufl, ufl.grad(u)) * v * ufl.dx
    L_gal = f_expr * v * ufl.dx
    
    # SUPG terms: add tau * (beta . grad(v)) * residual
    # residual = -eps*laplacian(u) + beta.grad(u) - f
    # For the bilinear form (trial function part):
    # SUPG test function modification: v_supg = tau * beta . grad(v)
    v_supg = tau * ufl.dot(beta_ufl, ufl.grad(v))
    
    # The SUPG contribution to bilinear form:
    # For P3 elements, we can include the diffusion term in the residual
    a_supg = (-eps_const * ufl.div(ufl.grad(u)) + ufl.dot(beta_ufl, ufl.grad(u))) * v_supg * ufl.dx
    L_supg = f_expr * v_supg * ufl.dx
    
    a_total = a_gal + a_supg
    L_total = L_gal + L_supg
    
    # Solve
    ksp_type = "gmres"
    pc_type = "ilu"
    rtol = 1e-12
    
    problem = petsc.LinearProblem(
        a_total, L_total, bcs=[bc],
        petsc_options={
            "ksp_type": ksp_type,
            "pc_type": pc_type,
            "ksp_rtol": str(rtol),
            "ksp_atol": "1e-14",
            "ksp_max_it": "5000",
            "ksp_monitor": None,
        },
        petsc_options_prefix="convdiff_"
    )
    u_sol = problem.solve()
    
    # Get iteration count
    ksp = problem.solver
    iterations = ksp.getIterationNumber()
    
    # Evaluate on 50x50 grid
    nx_out, ny_out = 50, 50
    xs = np.linspace(0, 1, nx_out)
    ys = np.linspace(0, 1, ny_out)
    XX, YY = np.meshgrid(xs, ys, indexing='ij')
    
    points = np.zeros((3, nx_out * ny_out))
    points[0, :] = XX.ravel()
    points[1, :] = YY.ravel()
    
    bb_tree = geometry.bb_tree(domain, domain.topology.dim)
    cell_candidates = geometry.compute_collisions_points(bb_tree, points.T)
    colliding_cells = geometry.compute_colliding_cells(domain, cell_candidates, points.T)
    
    points_on_proc = []
    cells_on_proc = []
    eval_map = []
    for i in range(points.shape[1]):
        links = colliding_cells.links(i)
        if len(links) > 0:
            points_on_proc.append(points.T[i])
            cells_on_proc.append(links[0])
            eval_map.append(i)
    
    u_values = np.full(nx_out * ny_out, np.nan)
    if len(points_on_proc) > 0:
        vals = u_sol.eval(np.array(points_on_proc), np.array(cells_on_proc, dtype=np.int32))
        u_values[eval_map] = vals.flatten()
    
    u_grid = u_values.reshape((nx_out, ny_out))
    
    solver_info = {
        "mesh_resolution": N,
        "element_degree": element_degree,
        "ksp_type": ksp_type,
        "pc_type": pc_type,
        "rtol": rtol,
        "iterations": int(iterations),
    }
    
    return {
        "u": u_grid,
        "solver_info": solver_info,
    }
```
