```python
import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry, nls
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc


def solve(case_spec: dict) -> dict:
    comm = MPI.COMM_WORLD
    
    # Parse case spec
    pde = case_spec.get("pde", {})
    domain_spec = case_spec.get("domain", {})
    time_spec = pde.get("time", {})
    
    # Parameters
    epsilon = pde.get("epsilon", 0.01)
    reaction_rho = pde.get("reaction_rho", 1.0)
    
    t_end = time_spec.get("t_end", 0.35)
    dt_val = time_spec.get("dt", 0.01)
    scheme = time_spec.get("scheme", "backward_euler")
    
    # Mesh
    mesh_resolution = 80
    element_degree = 1
    
    domain_obj = mesh.create_unit_square(comm, mesh_resolution, mesh_resolution, 
                                          cell_type=mesh.CellType.triangle)
    
    V = fem.functionspace(domain_obj, ("Lagrange", element_degree))
    
    # Spatial coordinate
    x = ufl.SpatialCoordinate(domain_obj)
    
    # Source term
    f_expr = (4.0 * ufl.exp(-200.0 * ((x[0] - 0.4)**2 + (x[1] - 0.6)**2))
              - 2.0 * ufl.exp(-200.0 * ((x[0] - 0.65)**2 + (x[1] - 0.35)**2)))
    
    # Reaction term: logistic R(u) = rho * u * (1 - u)
    # The PDE is: du/dt - eps * laplacian(u) + R(u) = f
    # where R(u) = rho * u * (1 - u) (logistic reaction)
    
    # Boundary conditions: u = g on boundary
    # Default: homogeneous or from case spec
    bc_spec = pde.get("boundary_conditions", {})
    
    # For this problem, let's check if there's a specific BC
    # Default: use initial condition value on boundary or zero
    tdim = domain_obj.topology.dim
    fdim = tdim - 1
    
    def boundary_all(x):
        return (np.isclose(x[0], 0.0) | np.isclose(x[0], 1.0) |
                np.isclose(x[1], 0.0) | np.isclose(x[1], 1.0))
    
    boundary_facets = mesh.locate_entities_boundary(domain_obj, fdim, boundary_all)
    dofs = fem.locate_dofs_topological(V, fdim, boundary_facets)
    
    # BC value: use initial condition on boundary
    u_bc = fem.Function(V)
    u_bc.interpolate(lambda x: 0.4 + 0.1 * np.sin(np.pi * x[0]) * np.sin(np.pi * x[1]))
    bc = fem.dirichletbc(u_bc, dofs)
    bcs = [bc]
    
    # Functions
    u_n = fem.Function(V)  # solution at previous time step
    u_n.interpolate(lambda x: 0.4 + 0.1 * np.sin(np.pi * x[0]) * np.sin(np.pi * x[1]))
    
    # Store initial condition for output
    u_initial_func = fem.Function(V)
    u_initial_func.x.array[:] = u_n.x.array[:]
    
    u = fem.Function(V)  # current solution (unknown)
    u.x.array[:] = u_n.x.array[:]
    
    v = ufl.TestFunction(V)
    
    dt = fem.Constant(domain_obj, PETSc.ScalarType(dt_val))
    eps_c = fem.Constant(domain_obj, PETSc.ScalarType(epsilon))
    rho_c = fem.Constant(domain_obj, PETSc.ScalarType(reaction_rho))
    
    # Backward Euler: (u - u_n)/dt - eps*laplacian(u) + rho*u*(1-u) = f
    # Weak form (residual):
    # (u - u_n)/dt * v dx + eps * grad(u) . grad(v) dx + rho * u * (1 - u) * v dx - f * v dx = 0
    
    F = ((u - u_n) / dt * v * ufl.dx
         + eps_c * ufl.inner(ufl.grad(u), ufl.grad(v)) * ufl.dx
         + rho_c * u * (1.0 - u) * v * ufl.dx
         - f_expr * v * ufl.dx)
    
    # Setup nonlinear problem
    problem = petsc.NonlinearProblem(F, u, bcs=bcs)
    solver = nls.petsc.NewtonSolver(comm, problem)
    solver.convergence_criterion = "incremental"
    solver.rtol = 1e-8
    solver.atol = 1e-10
    solver.max_it = 25
    
    ksp = solver.krylov_solver
    ksp.setType(PETSc.KSP.Type.GMRES)
    pc = ksp.getPC()
    pc.setType(PETSc.PC.Type.ILU)
    
    # Time stepping
    n_steps = int(np.round(t_end / dt_val))
    actual_dt = t_end / n_steps
    dt.value = actual_dt
    
    nonlinear_iterations = []
    total_linear_iterations = 0
    
    t = 0.0
    for step in range(n_steps):
        t += actual_dt
        
        # Update BC on boundary (IC-based BC stays constant since sin(pi*0)=0 on boundary)
        # Actually on the boundary of unit square, sin(pi*x)*sin(pi*y) = 0
        # So u_bc = 0.4 on boundary, which is constant. No update needed.
        
        n_iter, converged = solver.solve(u)
        assert converged, f"Newton solver did not converge at step {step}"
        
        nonlinear_iterations.append(n_iter)
        
        # Update previous solution
        u_n.x.array[:] = u.x.array[:]
    
    # Evaluate on 75x75 grid
    nx_out, ny_out = 75, 75
    xs = np.linspace(0.0, 1.0, nx_out)
    ys = np.linspace(0.0, 1.0, ny_out)
    XX, YY = np.meshgrid(xs, ys, indexing='ij')
    points = np.zeros((3, nx_out * ny_out))
    points[0, :] = XX.ravel()
    points[1, :] = YY.ravel()
    
    bb_tree = geometry.bb_tree(domain_obj, domain_obj.topology.dim)
    cell_candidates = geometry.compute_collisions_points(bb_tree, points.T)
    colliding_cells = geometry.compute_colliding_cells(domain_obj, cell_candidates, points.T)
    
    points_on_proc = []
    cells_on_proc = []
    eval_map = []
    for i in range(points.shape[1]):
        links = colliding_cells.links(i)
        if len(links) > 0:
            points_on_proc.append(points[:, i])
            cells_on_proc.append(links[0])
            eval_map.append(i)
    
    u_values = np.full(nx_out * ny_out, np.nan)
    if len(points_on_proc) > 0:
        pts_arr = np.array(points_on_proc)
        cells_arr = np.array(cells_on_proc, dtype=np.int32)
        vals = u.eval(pts_arr, cells_arr)
        u_values[eval_map] = vals.flatten()
    
    u_grid = u_values.reshape((nx_out, ny_out))
    
    # Also evaluate initial condition
    u_init_values = np.full(nx_out * ny_out, np.nan)
    if len(points_on_proc) > 0:
        vals_init = u_initial_func.eval(pts_arr, cells_arr)
        u_init_values[eval_map] = vals_init.flatten()
    
    u_init_grid = u_init_values.reshape((nx_out, ny_out))
    
    solver_info = {
        "mesh_resolution": mesh_resolution,
        "element_degree": element_degree,
        "ksp_type": "gmres",
        "pc_type": "ilu",
        "rtol": 1e-8,
        "dt": actual_dt,
        "n_steps": n_steps,
        "time_scheme": "backward_euler",
        "nonlinear_iterations": nonlinear_iterations,
    }
    
    return {
        "u": u_grid,
        "u_initial": u_init_grid,
        "solver_info": solver_info,
    }
```
