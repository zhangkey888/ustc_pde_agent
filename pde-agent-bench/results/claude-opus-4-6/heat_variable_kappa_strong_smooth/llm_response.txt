import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc


def solve(case_spec: dict) -> dict:
    comm = MPI.COMM_WORLD
    
    # Parameters
    nx = ny = 64
    degree = 2
    dt_val = 0.005
    t_end = 0.1
    n_steps = int(round(t_end / dt_val))
    
    # Create mesh
    domain = mesh.create_unit_square(comm, nx, ny, cell_type=mesh.CellType.triangle)
    tdim = domain.topology.dim
    fdim = tdim - 1
    
    # Function space
    V = fem.functionspace(domain, ("Lagrange", degree))
    
    # Spatial coordinate and time
    x = ufl.SpatialCoordinate(domain)
    
    # Time constant
    t = fem.Constant(domain, PETSc.ScalarType(0.0))
    dt = fem.Constant(domain, PETSc.ScalarType(dt_val))
    
    # Exact solution
    pi = ufl.pi
    u_exact_ufl = ufl.exp(-t) * ufl.sin(3 * pi * x[0]) * ufl.sin(2 * pi * x[1])
    
    # Variable kappa
    kappa_ufl = 1.0 + 0.8 * ufl.sin(2 * pi * x[0]) * ufl.sin(2 * pi * x[1])
    
    # Source term: f = du/dt - div(kappa * grad(u))
    # du/dt = -exp(-t)*sin(3*pi*x)*sin(2*pi*y)
    dudt = -ufl.exp(-t) * ufl.sin(3 * pi * x[0]) * ufl.sin(2 * pi * x[1])
    grad_u_exact = ufl.grad(u_exact_ufl)
    f_ufl = dudt - ufl.div(kappa_ufl * grad_u_exact)
    
    # Trial and test functions
    u = ufl.TrialFunction(V)
    v = ufl.TestFunction(V)
    
    # Previous solution
    u_n = fem.Function(V)
    
    # Interpolate initial condition at t=0
    t.value = 0.0
    u0_expr = fem.Expression(u_exact_ufl, V.element.interpolation_points)
    u_n.interpolate(u0_expr)
    
    # Store initial condition for output
    # Build grid for evaluation
    nx_grid, ny_grid = 50, 50
    xs = np.linspace(0, 1, nx_grid)
    ys = np.linspace(0, 1, ny_grid)
    XX, YY = np.meshgrid(xs, ys, indexing='ij')
    points_2d = np.zeros((3, nx_grid * ny_grid))
    points_2d[0] = XX.ravel()
    points_2d[1] = YY.ravel()
    
    # Backward Euler: (u - u_n)/dt - div(kappa * grad(u)) = f
    # Weak form: (u/dt)*v + kappa*grad(u)·grad(v) = (u_n/dt)*v + f*v
    a = (u / dt) * v * ufl.dx + kappa_ufl * ufl.inner(ufl.grad(u), ufl.grad(v)) * ufl.dx
    L = (u_n / dt) * v * ufl.dx + f_ufl * v * ufl.dx
    
    # Boundary conditions - all boundary
    boundary_facets = mesh.locate_entities_boundary(
        domain, fdim, lambda x: np.ones(x.shape[1], dtype=bool)
    )
    
    u_bc = fem.Function(V)
    bc_expr = fem.Expression(u_exact_ufl, V.element.interpolation_points)
    u_bc.interpolate(bc_expr)
    
    dofs = fem.locate_dofs_topological(V, fdim, boundary_facets)
    bc = fem.dirichletbc(u_bc, dofs)
    bcs = [bc]
    
    # Compile forms
    a_form = fem.form(a)
    L_form = fem.form(L)
    
    # Solution function
    u_sol = fem.Function(V)
    
    # Setup solver
    solver = PETSc.KSP().create(domain.comm)
    solver.setType(PETSc.KSP.Type.GMRES)
    pc = solver.getPC()
    pc.setType(PETSc.PC.Type.ILU)
    solver.setTolerances(rtol=1e-10, atol=1e-12, max_it=1000)
    
    total_iterations = 0
    
    # Evaluate initial condition on grid
    bb_tree = geometry.bb_tree(domain, tdim)
    cell_candidates = geometry.compute_collisions_points(bb_tree, points_2d.T)
    colliding_cells = geometry.compute_colliding_cells(domain, cell_candidates, points_2d.T)
    
    points_on_proc = []
    cells_on_proc = []
    eval_map = []
    for i in range(points_2d.shape[1]):
        links = colliding_cells.links(i)
        if len(links) > 0:
            points_on_proc.append(points_2d[:, i])
            cells_on_proc.append(links[0])
            eval_map.append(i)
    
    points_on_proc_arr = np.array(points_on_proc) if points_on_proc else np.empty((0, 3))
    cells_on_proc_arr = np.array(cells_on_proc, dtype=np.int32) if cells_on_proc else np.empty(0, dtype=np.int32)
    
    u_initial_vals = np.full(nx_grid * ny_grid, np.nan)
    if len(points_on_proc) > 0:
        vals = u_n.eval(points_on_proc_arr, cells_on_proc_arr)
        u_initial_vals[eval_map] = vals.flatten()
    u_initial = u_initial_vals.reshape((nx_grid, ny_grid))
    
    # Time stepping
    for step in range(n_steps):
        t.value = (step + 1) * dt_val
        
        # Update BC
        u_bc.interpolate(bc_expr)
        
        # Assemble matrix
        A = petsc.assemble_matrix(a_form, bcs=bcs)
        A.assemble()
        
        # Assemble RHS
        b = petsc.create_vector(L_form)
        with b.localForm() as loc:
            loc.set(0)
        petsc.assemble_vector(b, L_form)
        petsc.apply_lifting(b, [a_form], bcs=[bcs])
        b.ghostUpdate(addv=PETSc.InsertMode.ADD, mode=PETSc.ScatterMode.REVERSE)
        petsc.set_bc(b, bcs)
        
        # Solve
        solver.setOperators(A)
        solver.solve(b, u_sol.x.petsc_vec)
        u_sol.x.scatter_forward()
        
        total_iterations += solver.getIterationNumber()
        
        # Update previous solution
        u_n.x.array[:] = u_sol.x.array[:]
        
        # Clean up
        A.destroy()
        b.destroy()
    
    # Evaluate final solution on grid
    u_final_vals = np.full(nx_grid * ny_grid, np.nan)
    if len(points_on_proc) > 0:
        vals = u_sol.eval(points_on_proc_arr, cells_on_proc_arr)
        u_final_vals[eval_map] = vals.flatten()
    u_grid = u_final_vals.reshape((nx_grid, ny_grid))
    
    solver.destroy()
    
    return {
        "u": u_grid,
        "u_initial": u_initial,
        "solver_info": {
            "mesh_resolution": nx,
            "element_degree": degree,
            "ksp_type": "gmres",
            "pc_type": "ilu",
            "rtol": 1e-10,
            "iterations": total_iterations,
            "dt": dt_val,
            "n_steps": n_steps,
            "time_scheme": "backward_euler",
        }
    }
