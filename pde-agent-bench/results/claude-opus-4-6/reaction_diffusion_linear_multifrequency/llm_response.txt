```python
import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc

ScalarType = PETSc.ScalarType


def solve(case_spec: dict) -> dict:
    comm = MPI.COMM_WORLD

    # Parse parameters
    pde = case_spec.get("pde", {})
    time_params = pde.get("time", {})
    t_end = time_params.get("t_end", 0.4)
    dt_suggested = time_params.get("dt", 0.005)
    scheme = time_params.get("scheme", "crank_nicolson")

    # Choose mesh resolution and element degree for accuracy
    # The manufactured solution has high-frequency components (6*pi*x, 5*pi*y)
    # Need sufficient resolution to capture these
    N = 100  # mesh resolution
    degree = 2  # quadratic elements for better accuracy
    dt = dt_suggested

    # Create mesh
    domain = mesh.create_unit_square(comm, N, N, cell_type=mesh.CellType.triangle)
    V = fem.functionspace(domain, ("Lagrange", degree))

    # Spatial coordinates and time
    x = ufl.SpatialCoordinate(domain)
    t = fem.Constant(domain, ScalarType(0.0))

    # Manufactured solution: u = exp(-t)*(sin(pi*x)*sin(pi*y) + 0.2*sin(6*pi*x)*sin(5*pi*y))
    pi = ufl.pi
    u_exact_ufl = ufl.exp(-t) * (
        ufl.sin(pi * x[0]) * ufl.sin(pi * x[1])
        + 0.2 * ufl.sin(6 * pi * x[0]) * ufl.sin(5 * pi * x[1])
    )

    # Diffusion coefficient epsilon
    epsilon_val = pde.get("coefficients", {}).get("epsilon", 1.0)
    epsilon = fem.Constant(domain, ScalarType(epsilon_val))

    # Reaction term R(u) - for linear reaction-diffusion, R(u) = r*u
    # Check if there's a reaction coefficient
    r_val = pde.get("coefficients", {}).get("reaction", 0.0)
    r = fem.Constant(domain, ScalarType(r_val))

    # Compute source term f from manufactured solution:
    # du/dt - epsilon * laplacian(u) + r*u = f
    # du/dt = -exp(-t)*(sin(pi*x)*sin(pi*y) + 0.2*sin(6*pi*x)*sin(5*pi*y)) = -u_exact
    # laplacian(u) = exp(-t)*(-pi^2 - pi^2)*sin(pi*x)*sin(pi*y) + 0.2*exp(-t)*(-(6*pi)^2 - (5*pi)^2)*sin(6*pi*x)*sin(5*pi*y)
    #              = exp(-t)*(-2*pi^2*sin(pi*x)*sin(pi*y) - 0.2*61*pi^2*sin(6*pi*x)*sin(5*pi*y))
    # f = du/dt - epsilon*laplacian(u) + r*u

    # Let UFL compute the source term symbolically
    # du/dt for the manufactured solution
    du_dt_ufl = -ufl.exp(-t) * (
        ufl.sin(pi * x[0]) * ufl.sin(pi * x[1])
        + 0.2 * ufl.sin(6 * pi * x[0]) * ufl.sin(5 * pi * x[1])
    )

    # Laplacian of u_exact
    grad_u_exact = ufl.grad(u_exact_ufl)
    laplacian_u_exact = ufl.div(grad_u_exact)

    f_ufl = du_dt_ufl - epsilon * laplacian_u_exact + r * u_exact_ufl

    # Functions
    u_n = fem.Function(V, name="u_n")  # solution at previous time step
    u_h = fem.Function(V, name="u_h")  # solution at current time step

    # Trial and test functions
    u = ufl.TrialFunction(V)
    v = ufl.TestFunction(V)

    dt_const = fem.Constant(domain, ScalarType(dt))

    # Crank-Nicolson scheme:
    # (u - u_n)/dt - epsilon * 0.5*(laplacian(u) + laplacian(u_n)) + r * 0.5*(u + u_n) = 0.5*(f^{n+1} + f^n)
    # Rearranged as bilinear + linear form

    # Time at n+1 and n
    t_np1 = fem.Constant(domain, ScalarType(0.0))
    t_n_const = fem.Constant(domain, ScalarType(0.0))

    # Source at t^{n+1} and t^n
    # We'll use fem.Expression to evaluate f at different times
    # But it's easier to define f as a function of t (the Constant) and update t

    if scheme == "crank_nicolson":
        theta = 0.5
    elif scheme == "backward_euler":
        theta = 1.0
    else:
        theta = 0.5  # default to CN

    # Bilinear form (LHS)
    a_form = (
        ufl.inner(u, v) * ufl.dx
        + dt_const * theta * epsilon * ufl.inner(ufl.grad(u), ufl.grad(v)) * ufl.dx
        + dt_const * theta * r * ufl.inner(u, v) * ufl.dx
    )

    # For the RHS, we need f at t^{n+1} and t^n
    # We'll compute f_np1 and f_n by setting t to appropriate values
    # But since t is a Constant in the UFL expression, we need two separate expressions
    # or update t and reassemble

    # Alternative approach: precompute f at each time step
    # Let's use the approach of updating t and assembling

    # RHS linear form (depends on t through f_ufl and u_exact for BC)
    # f at current t (which we'll set to t^{n+1} for f_np1 evaluation)
    # We need f at both t_n and t_{n+1}

    # Strategy: use two Constants for time, build two f expressions
    t_old = fem.Constant(domain, ScalarType(0.0))
    t_new = fem.Constant(domain, ScalarType(0.0))

    u_exact_new = ufl.exp(-t_new) * (
        ufl.sin(pi * x[0]) * ufl.sin(pi * x[1])
        + 0.2 * ufl.sin(6 * pi * x[0]) * ufl.sin(5 * pi * x[1])
    )
    u_exact_old = ufl.exp(-t_old) * (
        ufl.sin(pi * x[0]) * ufl.sin(pi * x[1])
        + 0.2 * ufl.sin(6 * pi * x[0]) * ufl.sin(5 * pi * x[1])
    )

    du_dt_new = -u_exact_new
    du_dt_old = -u_exact_old

    grad_u_exact_new = ufl.grad(u_exact_new)
    laplacian_u_exact_new = ufl.div(grad_u_exact_new)
    f_new = du_dt_new - epsilon * laplacian_u_exact_new + r * u_exact_new

    grad_u_exact_old = ufl.grad(u_exact_old)
    laplacian_u_exact_old = ufl.div(grad_u_exact_old)
    f_old = du_dt_old - epsilon * laplacian_u_exact_old + r * u_exact_old

    # RHS
    L_form = (
        ufl.inner(u_n, v) * ufl.dx
        - dt_const * (1 - theta) * epsilon * ufl.inner(ufl.grad(u_n), ufl.grad(v)) * ufl.dx
        - dt_const * (1 - theta) * r * ufl.inner(u_n, v) * ufl.dx
        + dt_const * (theta * ufl.inner(f_new, v) * ufl.dx + (1 - theta) * ufl.inner(f_old, v) * ufl.dx)
    )

    # Compile forms
    a_compiled = fem.form(a_form)
    L_compiled = fem.form(L_form)

    # Boundary conditions - u = u_exact on all boundaries
    tdim = domain.topology.dim
    fdim = tdim - 1

    boundary_facets = mesh.locate_entities_boundary(
        domain, fdim, lambda x: np.ones(x.shape[1], dtype=bool)
    )

    bc_func = fem.Function(V)
    dofs = fem.locate_dofs_topological(V, fdim, boundary_facets)

    # Set initial condition
    t_old.value = 0.0
    t_new.value = 0.0

    # Interpolate initial condition
    u_exact_expr_init = fem.Expression(u_exact_old, V.element.interpolation_points)
    u_n.interpolate(u_exact_expr_init)

    # Store initial condition for output
    nx_out, ny_out = 80, 80
    xs = np.linspace(0, 1, nx_out)
    ys = np.linspace(0, 1, ny_out)
    X, Y = np.meshgrid(xs, ys, indexing='ij')
    points_2d = np.vstack([X.ravel(), Y.ravel(), np.zeros(nx_out * ny_out)])

    # Assemble matrix (constant in time)
    A = petsc.assemble_matrix(a_compiled, bcs=[fem.dirichletbc(bc_func, dofs)])
    A.assemble()

    # Setup solver
    solver = PETSc.KSP().create(domain.comm)
    solver.setOperators(A)
    solver.setType(PETSc.KSP.Type.GMRES)
    pc = solver.getPC()
    pc.setType(PETSc.PC.Type.ILU)
    solver.setTolerances(rtol=1e-10, atol=1e-12, max_it=1000)

    # Create RHS vector
    b = petsc.create_vector(L_compiled)

    # Time stepping
    n_steps = int(np.round(t_end / dt))
    dt = t_end / n_steps  # adjust dt to hit t_end exactly
    dt_const.value = dt

    # Reassemble A since dt might have changed
    A.zeroEntries()
    petsc.assemble_matrix(A, a_compiled, bcs=[fem.dirichletbc(bc_func, dofs)])
    A.assemble()

    total_iterations = 0

    # Expression for BC update
    bc_expr = fem.Expression(u_exact_new, V.element.interpolation_points)

    for step in range(n_steps):
        t_current = step * dt
        t_next = (step + 1) * dt

        t_old.value = t_current
        t_new.value = t_next

        # Update boundary condition
        bc_func.interpolate(bc_expr)
        bc = fem.dirichletbc(bc_func, dofs)

        # Assemble RHS
        with b.localForm() as loc:
            loc.set(0)
        petsc.assemble_vector(b, L_compiled)
        petsc.apply_lifting(b, [a_compiled], bcs=[[bc]])
        b.ghostUpdate(addv=PETSc.InsertMode.ADD, mode=PETSc.ScatterMode.REVERSE)
        petsc.set_bc(b, [bc])

        # Solve
        solver.solve(b, u_h.x.petsc_vec)
        u_h.x.scatter_forward()

        total_iterations += solver.getIterationNumber()

        # Update for next step
        u_n.x.array[:] = u_h.x.array[:]

    # Evaluate solution on output grid
    bb_tree = geometry.bb_tree(domain, domain.topology.dim)
    cell_candidates = geometry.compute_collisions_points(bb_tree, points_2d.T)
    colliding_cells = geometry.compute_colliding_cells(domain, cell_candidates, points_2d.T)

    points_on_proc = []
    cells_on_proc = []
    eval_map = []
    for i in range(points_2d.shape[1]):
        links = colliding_cells.links(i)
        if len(links) > 0:
            points_on_proc.append(points_2d.T[i])
            cells_on_proc.append(links[0])
            eval_map.append(i)

    u_values = np.full(points_2d.shape[1], np.nan)
    if len(points_on_proc) > 0:
        vals = u_h.eval(np.array(points_on_proc), np.array(cells_on_proc, dtype=np.int32))
        u_values[eval_map] = vals.flatten()

    u_grid = u_values.reshape((nx_out, ny_out))

    # Also evaluate initial condition for output
    # Reset time to 0 for initial condition
    t_old.value = 0.0
    u_init_func = fem.Function(V)
    u_exact_expr_init2 = fem.Expression(u_exact_old, V.element.interpolation_points)
    u_init_func.interpolate(u_exact_expr_init2)

    u_init_values = np.full(points_2d.shape[1], np.nan)
    if len(points_on_proc) > 0:
        vals_init = u_init_func.eval(np.array(points_on_proc), np.array(cells_on_proc, dtype=np.int32))
        u_init_values[eval_map] = vals_init.flatten()

    u_initial_grid = u_init_values.reshape((nx_out, ny_out))

    # Cleanup
    solver.destroy()
    A.destroy()
    b.destroy()

    time_scheme_name = "crank_nicolson" if theta == 0.5 else "backward_euler"

    return {
        "u": u_grid,
        "u_initial": u_initial_grid,
        "solver_info": {
            "mesh_resolution": N,
            "element_degree": degree,
            "ksp_type": "gmres",
            "pc_type": "ilu",
            "rtol": 1e-10,
            "iterations": total_iterations,
            "dt": dt,
            "n_steps": n_steps,
            "time_scheme": time_scheme_name,
        },
    }
```
