```python
import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, nls, geometry
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc

ScalarType = PETSc.ScalarType


def solve(case_spec: dict) -> dict:
    comm = MPI.COMM_WORLD
    
    # Parameters
    nu_val = case_spec.get("pde", {}).get("viscosity", 0.22)
    mesh_res = 80
    degree_u = 2
    degree_p = 1
    newton_rtol = 1e-8
    newton_atol = 1e-10
    newton_max_it = 50
    
    # Create mesh
    domain = mesh.create_unit_square(comm, mesh_res, mesh_res, cell_type=mesh.CellType.triangle)
    tdim = domain.topology.dim
    fdim = tdim - 1
    
    # Function spaces (Taylor-Hood P2/P1)
    vel_elem = ("Lagrange", degree_u, (domain.geometry.dim,))
    pres_elem = ("Lagrange", degree_p)
    
    V = fem.functionspace(domain, vel_elem)
    Q = fem.functionspace(domain, pres_elem)
    
    # Mixed function space
    from dolfinx.fem import functionspace
    from basix.ufl import element, mixed_element
    
    vel_el = element("Lagrange", domain.basix_cell(), degree_u, shape=(domain.geometry.dim,))
    pres_el = element("Lagrange", domain.basix_cell(), degree_p)
    mixed_el = mixed_element([vel_el, pres_el])
    
    W = functionspace(domain, mixed_el)
    
    # Solution function
    w = fem.Function(W)
    (u, p) = ufl.split(w)
    (v, q) = ufl.TestFunctions(W)
    
    # Viscosity
    nu = fem.Constant(domain, ScalarType(nu_val))
    
    # Source term
    f = fem.Constant(domain, ScalarType((0.0, 0.0)))
    
    # Weak form (residual)
    F = (
        nu * ufl.inner(ufl.grad(u), ufl.grad(v)) * ufl.dx
        + ufl.inner(ufl.grad(u) * u, v) * ufl.dx
        - p * ufl.div(v) * ufl.dx
        + q * ufl.div(u) * ufl.dx
        - ufl.inner(f, v) * ufl.dx
    )
    
    # Parse boundary conditions from case_spec
    bc_specs = case_spec.get("bcs", [])
    
    bcs = []
    
    # We need sub-spaces for BCs
    W0, W0_map = W.sub(0).collapse()
    
    for bc_spec in bc_specs:
        bc_type = bc_spec.get("type", "")
        location = bc_spec.get("location", "")
        value = bc_spec.get("value", None)
        
        if bc_type == "dirichlet":
            # Determine boundary marker
            if location == "left":
                marker = lambda x: np.isclose(x[0], 0.0)
            elif location == "right":
                marker = lambda x: np.isclose(x[0], 1.0)
            elif location == "bottom":
                marker = lambda x: np.isclose(x[1], 0.0)
            elif location == "top":
                marker = lambda x: np.isclose(x[1], 1.0)
            elif location == "all":
                marker = lambda x: (np.isclose(x[0], 0.0) | np.isclose(x[0], 1.0) |
                                     np.isclose(x[1], 0.0) | np.isclose(x[1], 1.0))
            else:
                continue
            
            facets = mesh.locate_entities_boundary(domain, fdim, marker)
            dofs = fem.locate_dofs_topological((W.sub(0), W0), fdim, facets)
            
            # Parse value
            if value is not None:
                if isinstance(value, (list, tuple)) and len(value) == 2:
                    # Could be expressions or constants
                    val_strs = [str(v_) for v_ in value]
                    
                    # Try to create interpolation function
                    u_bc = fem.Function(W0)
                    
                    # Build a lambda for interpolation
                    def make_bc_func(val_strs):
                        def bc_func(x):
                            result = np.zeros((domain.geometry.dim, x.shape[1]))
                            for i, vs in enumerate(val_strs):
                                # Replace common math expressions
                                expr = vs.replace("^", "**")
                                # Create namespace for eval
                                ns = {"x": x, "np": np, "pi": np.pi, "sin": np.sin, 
                                      "cos": np.cos, "exp": np.exp, "sqrt": np.sqrt}
                                # x[0], x[1] are available
                                try:
                                    result[i] = eval(expr, ns)
                                except:
                                    result[i] = float(expr) * np.ones(x.shape[1])
                            return result
                        return bc_func
                    
                    u_bc.interpolate(make_bc_func(val_strs))
                    bc = fem.dirichletbc(u_bc, dofs, W.sub(0))
                    bcs.append(bc)
                else:
                    # Scalar or other
                    u_bc = fem.Function(W0)
                    u_bc.interpolate(lambda x: np.zeros((domain.geometry.dim, x.shape[1])))
                    bc = fem.dirichletbc(u_bc, dofs, W.sub(0))
                    bcs.append(bc)
            else:
                # Zero BC
                u_bc = fem.Function(W0)
                u_bc.interpolate(lambda x: np.zeros((domain.geometry.dim, x.shape[1])))
                bc = fem.dirichletbc(u_bc, dofs, W.sub(0))
                bcs.append(bc)
    
    # If no BCs parsed, set up default (all boundaries with zero velocity)
    if len(bcs) == 0:
        facets = mesh.locate_entities_boundary(
            domain, fdim, 
            lambda x: np.ones(x.shape[1], dtype=bool)
        )
        dofs = fem.locate_dofs_topological((W.sub(0), W0), fdim, facets)
        u_bc = fem.Function(W0)
        u_bc.interpolate(lambda x: np.zeros((domain.geometry.dim, x.shape[1])))
        bc = fem.dirichletbc(u_bc, dofs, W.sub(0))
        bcs.append(bc)
    
    # Initial guess: zero (for moderate viscosity this should work)
    w.x.array[:] = 0.0
    
    # Apply BC values to initial guess
    # Interpolate BCs into w
    for bc in bcs:
        bc.set(w.x.array)
    w.x.scatter_forward()
    
    # Solve nonlinear problem
    problem = petsc.NonlinearProblem(F, w, bcs=bcs)
    solver = nls.petsc.NewtonSolver(comm, problem)
    
    solver.convergence_criterion = "incremental"
    solver.rtol = newton_rtol
    solver.atol = newton_atol
    solver.max_it = newton_max_it
    solver.relaxation_parameter = 1.0
    
    ksp = solver.krylov_solver
    ksp.setType(PETSc.KSP.Type.GMRES)
    ksp.setTolerances(rtol=1e-8, max_it=1000)
    pc = ksp.getPC()
    pc.setType(PETSc.PC.Type.LU)
    pc.setFactorSolverType("mumps")
    
    n_newton, converged = solver.solve(w)
    assert converged, f"Newton solver did not converge after {n_newton} iterations"
    w.x.scatter_forward()
    
    # Extract velocity sub-function
    u_sol = w.sub(0).collapse()
    
    # Evaluate on 50x50 grid
    nx_eval, ny_eval = 50, 50
    xs = np.linspace(0.0, 1.0, nx_eval)
    ys = np.linspace(0.0, 1.0, ny_eval)
    XX, YY = np.meshgrid(xs, ys, indexing='ij')
    
    points = np.zeros((3, nx_eval * ny_eval))
    points[0, :] = XX.ravel()
    points[1, :] = YY.ravel()
    
    # Point evaluation
    bb_tree = geometry.bb_tree(domain, tdim)
    cell_candidates = geometry.compute_collisions_points(bb_tree, points.T)
    colliding_cells = geometry.compute_colliding_cells(domain, cell_candidates, points.T)
    
    points_on_proc = []
    cells_on_proc = []
    eval_map = []
    
    for i in range(points.shape[1]):
        links = colliding_cells.links(i)
        if len(links) > 0:
            points_on_proc.append(points[:, i])
            cells_on_proc.append(links[0])
            eval_map.append(i)
    
    vel_magnitude = np.full(nx_eval * ny_eval, np.nan)
    
    if len(points_on_proc) > 0:
        pts_arr = np.array(points_on_proc)
        cells_arr = np.array(cells_on_proc, dtype=np.int32)
        vals = u_sol.eval(pts_arr, cells_arr)
        # vals shape: (n_points, gdim)
        vel_mag = np.sqrt(vals[:, 0]**2 + vals[:, 1]**2)
        for idx, global_idx in enumerate(eval_map):
            vel_magnitude[global_idx] = vel_mag[idx]
    
    u_grid = vel_magnitude.reshape((nx_eval, ny_eval))
    
    # Get total linear iterations
    total_linear_its = ksp.getIterationNumber()
    
    result = {
        "u": u_grid,
        "solver_info": {
            "mesh_resolution": mesh_res,
            "element_degree": degree_u,
            "ksp_type": "gmres",
            "pc_type": "lu",
            "rtol": newton_rtol,
            "iterations": total_linear_its,
            "nonlinear_iterations": [int(n_newton)],
        }
    }
    
    return result
```
